        // GroupDocs.Merger APIን በመጠቀም CSV ፋይልን ክፈሉ።

        string filePathOut = "output.csv";

        // የSplitOptions ክፍልን በውጤት ፋይሎች ዱካ ቅርጸት አስጀምር

            merger.Split(splitOptions);
          }
